app.post("/PassportAuthentication/login", passport.authenticate('local'), function (req, res) {
    var user = req.user;
    res.json(user);
});

The Logout operation is as below

app.post("/PassportAuthentication/logout", function (req, res) {
    //req.logout();
    req.session.destroy();
    res.send(200);
});
