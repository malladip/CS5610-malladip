This experiment is about locally storing data in browser

Code:

function saveName() {
    if (typeof (Storage) != "undefined") {
        if (localStorage.name) {
            localStorage.name = document.getElementById("name").value;
        } else {
            localStorage.name = "No name entered";
        }
        document.getElementById("message").innerHTML = "Welcome back " + localStorage.name;
        $("#message").dialog();
    } else {
        document.getElementById("message").innerHTML = "Your browser does not support web storage!!";
        $("#message").dialog();
    }
}
function loadName() {
    if (localStorage.name != null) {
        document.getElementById("message").innerHTML = "Welcome back " + localStorage.name;
        $("#message").dialog();
    }
}
